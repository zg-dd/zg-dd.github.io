一、Milvus 与关系型数据库概念对比

Milvus 概念关系型数据库概念说明
DatabaseDatabase数据库，最顶层的数据组织单位
CollectionTable集合相当于表，存储具有相同结构的数据
FieldColumn字段相当于列，定义数据的属性
EntityRow实体相当于行，表示一条完整的数据记录
SchemaTable Schema模式定义了数据结构和约束
Vector Field-向量字段是 Milvus 特有的，用于存储嵌入向量
Primary KeyPrimary Key主键，唯一标识每条记录

二、连接 Milvus 服务器

2.1 基本连接

from pymilvus import MilvusClient
import yaml
import os

# 读取配置文件
config_path = os.path.join(os.path.dirname(__file__), 'config.yaml')
with open(config_path, 'r', encoding='utf-8') as f:
    config = yaml.safe_load(f)

# 创建客户端连接
client = MilvusClient(
    uri=config['milvus']['uri'],      # Milvus 服务器地址
    token=config['milvus']['token']   # 认证令牌
)

2.2 配置文件示例 (config.yaml)

milvus:
  uri: "http://localhost:19530"  # 本地部署
  # uri: "https://your-cluster.zillizcloud.com"  # 云服务
  token: "root:Milvus"  # 用户名:密码

三、数据库基础操作

3.1 创建数据库

# 创建数据库
client.create_database(db_name="my_database_1")

# 对比 MySQL
# CREATE DATABASE my_database_1;

3.2 查看数据库列表

# 获取所有数据库
dblist = client.list_databases()
print(f"数据库列表: {dblist}")

# 对比 MySQL
# SHOW DATABASES;

输出示例：

数据库列表: ['default', 'my_database_1']

3.3 查看数据库详情

# 获取数据库详细信息
try:
    db_info = client.describe_database("my_database_1")
    print(f"数据库详情: {db_info}")
except Exception as e:
    print(f"获取失败: {e}")

# 对比 MySQL
# SELECT * FROM information_schema.SCHEMATA WHERE SCHEMA_NAME = 'my_database_1';

3.4 切换数据库

# 切换到指定数据库
client.use_database("my_database_1")

# 对比 MySQL
# USE my_database_1;

3.5 删除数据库

# 删除数据库
client.drop_database("my_database_1")

# 对比 MySQL
# DROP DATABASE my_database_1;

四、数据库属性管理

4.1 可修改的数据库属性

属性名称说明默认值对比关系型数据库
database.replica.number副本数量1类似 MySQL 的主从复制配置
timezone默认时区设置UTC类似 MySQL 的 time_zone 变量
database.diskQuota.mb最大磁盘空间(MB)无限制类似 MySQL 的表空间配额
database.max.collections最大集合数量无限制类似 MySQL 的最大表数量限制
database.force.deny.writing强制拒绝写操作false类似 MySQL 的 read_only 模式
database.force.deny.reading强制拒绝读操作falseMySQL 中较少使用此功能

4.2 修改数据库属性

# 修改多个属性
client.alter_database_properties(
    "my_database_1",
    {
        "database.replica.number": 2,           # 设置2个副本
        "timezone": "Asia/Shanghai",            # 设置时区为上海
        "database.diskQuota.mb": 10240,         # 限制磁盘空间为10GB
        "database.max.collections": 100         # 最多100个集合
    }
)

# 对比 MySQL
# ALTER DATABASE my_database_1 CHARACTER SET utf8mb4;
# SET GLOBAL time_zone = 'Asia/Shanghai';

4.3 删除数据库属性

# 删除指定属性（恢复为默认值）
client.drop_database_properties(
    "my_database_1",
    ["timezone", "database.diskQuota.mb"]
)

# 注意：关系型数据库通常不支持"删除"属性，只能重置为默认值

五、完整操作示例

from pymilvus import MilvusClient
import yaml
import os

def database_operations_demo():
    """数据库操作完整示例"""

    # 1. 连接服务器
    print("=== 连接 Milvus 服务器 ===")
    config_path = os.path.join(os.path.dirname(__file__), 'config.yaml')
    with open(config_path, 'r', encoding='utf-8') as f:
        config = yaml.safe_load(f)

    client = MilvusClient(
        uri=config['milvus']['uri'],
        token=config['milvus']['token']
    )
    print("✅ 连接成功")

    # 2. 创建数据库
    print("\n=== 创建数据库 ===")
    db_name = "product_database"
    client.create_database(db_name=db_name)
    print(f"✅ 数据库 '{db_name}' 创建成功")

    # 3. 查看数据库列表
    print("\n=== 数据库列表 ===")
    databases = client.list_databases()
    for db in databases:
        print(f"  - {db}")

    # 4. 配置数据库属性
    print("\n=== 配置数据库属性 ===")
    client.alter_database_properties(
        db_name,
        {
            "database.replica.number": 2,
            "timezone": "Asia/Shanghai",
            "database.max.collections": 50
        }
    )
    print("✅ 属性配置完成")

    # 5. 切换数据库
    print("\n=== 切换数据库 ===")
    client.use_database(db_name)
    print(f"✅ 已切换到 '{db_name}'")

    # 6. 查看数据库详情
    print("\n=== 数据库详情 ===")
    try:
        db_info = client.describe_database(db_name)
        print(f"  {db_info}")
    except Exception as e:
        print(f"  获取详情失败: {e}")

    # 7. 清理：删除数据库
    print("\n=== 清理测试数据 ===")
    client.drop_database(db_name)
    print(f"✅ 数据库 '{db_name}' 已删除")

if __name__ == "__main__":
    database_operations_demo()

六、最佳实践与注意事项

6.1 数据库命名规范

# ✅ 推荐的命名方式
"user_vectors"
"product_embeddings"
"image_search_db"

# ❌ 避免的命名方式
"Database-1"        # 避免使用连字符
"数据库1"           # 避免使用非ASCII字符
"db with spaces"   # 避免使用空格

6.2 副本配置建议

场景推荐副本数说明
开发测试环境1节省资源
生产环境（低负载）2基本高可用
生产环境（高负载）3+高可用 + 负载均衡
# 生产环境配置示例
client.alter_database_properties(
    "production_db",
    {
        "database.replica.number": 3,              # 3个副本
        "database.diskQuota.mb": 102400,           # 100GB限制
        "database.max.collections": 200,           # 最多200个集合
        "database.force.deny.writing": False       # 允许写入
    }
)

6.3 时区处理

# 设置数据库时区
client.alter_database_properties(
    "my_db",
    {"timezone": "Asia/Shanghai"}
)

# 常用时区
# "UTC"              # 协调世界时
# "Asia/Shanghai"    # 中国标准时间
# "America/New_York" # 美国东部时间
# "Europe/London"    # 英国时间

6.4 错误处理

from pymilvus import MilvusException

try:
    client.create_database("test_db")
except MilvusException as e:
    if "already exists" in str(e):
        print("数据库已存在，跳过创建")
    else:
        print(f"创建失败: {e}")
        raise

八、关键要点总结

  1. 连接管理：Milvus 使用 URI + Token 方式连接，比 MySQL 的主机+端口+用户名+密码更简洁
  2. 数据库隔离：每个数据库可以独立配置副本数、时区、配额等属性
  3. 默认数据库：Milvus 有一个名为 default 的默认数据库，类似 MySQL 的 mysql 系统数据库
  4. 属性动态修改：可以在运行时动态修改数据库属性，无需重启服务
  5. 副本机制：Milvus 的副本机制是内置的，不需要像 MySQL 那样手动配置主从复制
  6. 时区感知：Milvus 对时区的支持更加原生，特别是 TIMESTAMPTZ 字段类型
